function solution(s, n) {
    return s.split('').reduce((acc, cur) => {
        if (cur === ' ') return acc + cur;
        const code = cur.charCodeAt(0);
        const newCode = code + n > (code > 90 ? 122 : 90) ? code - (26 - n) : code + n;
        return acc + String.fromCharCode(newCode);
    },'');
}
