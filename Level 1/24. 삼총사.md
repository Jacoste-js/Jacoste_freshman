function solution(number) {
    const trio = [];
    const findTrio = (left, right) => {
        if (right.length === 3) trio.push(right);
        else left.forEach((el, i) => findTrio(left.slice(i + 1), right.concat(el)));
    }
    findTrio(number, []);
    return trio.filter((el) => el[0] + el[1] + el[2] === 0).length;
}
