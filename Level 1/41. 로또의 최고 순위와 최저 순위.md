function solution(lottos, win_nums) {
    let min = win_nums.reduce((acc, cur) => lottos.includes(cur) ? ++acc : acc, 0);
    let zero = lottos.reduce((acc, cur) => cur === 0 ? ++acc : acc, 0);
    if (min === 0 && zero === 0) return [6, 6];
    return [7 - (min + zero), 7 - (min ? min : 1)];
}
