function solution(food) {
    let ret = '0';
    for (let str, i = food.length - 1; i > 0; i--) {
        str = i.toString().repeat(parseInt(food[i] / 2));
        ret = str.concat(ret, str);
    }
    return ret;
}

function solution(food) {
    let str = '';
    food.forEach((el, i) => str += `${i}`.repeat(el % 2 ? (el - 1) / 2 : el / 2))
    return str + '0' + str.split('').reverse().join('');
}
