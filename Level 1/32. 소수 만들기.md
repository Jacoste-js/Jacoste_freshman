function solution(nums) {
    let ret = 0;
    let i, sqrt;
    
    function prime(num) {
        if (num === 7) {
            ret++;
            return ;
        }
        else if (num < 11) return ;
        sqrt = Math.sqrt(num);
        for (i = 3; i <= sqrt; i += 2)
            if (!(num % i)) break;
        if (sqrt < i) ret++;
    }
    
    function rec(arr, sum, cnt){
        if (cnt === 3){
            if (sum % 2) prime(sum);
            return;
        }
        arr.forEach((el, idx) =>{
            rec(arr.filter((e,i) => idx < i), sum + el, cnt + 1);   
        })
    }
    rec(nums, 0, 0);
    
    return ret;
}
