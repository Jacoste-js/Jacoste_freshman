function isPrime(n) {
    if ((n > 2 && n % 2 === 0) || n <= 1) return 0;
    for (let i = 3; i <= Math.sqrt(n); i += 2)
        if (n % i === 0) return 0;
    return 1;
}

function solution(nums) {
    const getSum = (cnt, num, idx) => {
        let prime = 0;
        if (cnt === 3) return isPrime(num);
        else {
            for (let i = idx; i < nums.length; i++)
                prime += getSum(cnt + 1, num + nums[i], i + 1);
        }
        return prime;
    }
    return getSum(0, 0, 0);
}

function solution(nums) {
    let ret = 0;
    let i, sqrt;
    
    function prime(num) {
        if (num === 7) {
            ret++;
            return ;
        }
        else if (num < 11) return ;
        sqrt = Math.sqrt(num);
        for (i = 3; i <= sqrt; i += 2)
            if (!(num % i)) break;
        if (sqrt < i) ret++;
    }
    
    function rec(arr, sum, cnt){
        if (cnt === 3){
            if (sum % 2) prime(sum);
            return;
        }
        arr.forEach((el, idx) =>{
            rec(arr.filter((e,i) => idx < i), sum + el, cnt + 1);   
        })
    }
    rec(nums, 0, 0);
    
    return ret;
}
