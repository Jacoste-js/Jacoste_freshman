function solution(sizes) {
    const [w, h] = sizes.reduce((acc, cur) => {
        return [Math.max(acc[0], Math.min(...cur)), Math.max(acc[1], Math.max(...cur))];
    }, [0, 0]);
    return w * h;
}

function solution(sizes) {
    let ret = [0,0];
    sizes.forEach((el) =>{
        ret[0] = Math.max(ret[0], Math.max(el[0], el[1]))
        ret[1] = Math.max(ret[1], Math.min(el[0], el[1]))
    })
    return (ret[0] * ret[1])
}
