function solution(n, m) {
    let ret = [Math.min(n, m) + 1, Math.max(n, m) - 1];
    
    while (--ret[0] > 0) if (!(n % ret[0]) && !(m % ret[0])) break;
    while (++ret[1]) if (!(ret[1] % n) && !(ret[1] % m)) break;
    return ret;
}

function solution(n, m) {
    let pro = 1;
    for (let i = 2; i <= n;)
    {
        if (n % i === 0 && m % i === 0)
        {
            n /= i;
            m /= i;
            pro *= i;
        }
        else
            i++;
    }
    return [pro, n * m * pro];
}
