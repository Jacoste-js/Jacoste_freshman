function solution(n, m) {
    let pro = 1;
    for (let i = 2; i <= n;)
    {
        if (n % i === 0 && m % i === 0)
        {
            n /= i;
            m /= i;
            pro *= i;
        }
        else
            i++;
    }
    return [pro, n * m * pro];
}
