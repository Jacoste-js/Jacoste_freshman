function solution(n) {
    let cnt = 4;
    let j, sqrt;
    if (n === 2) return 1;
    else if (n === 3) return 2;
    else if (n <= 6) return 3;
    else if (n <= 10) return 4;
    for (let i = 11; i <= n; i += 2){
        sqrt = Math.sqrt(i);
        for (j = 3; j <= sqrt; j += 2)
            if (!(i % j)) break;
        if (j > sqrt) cnt++;
    }
    return cnt;
}
