function solution(N, stages) {
    let obj = {};
    let len = stages.length;
    let tmp;
    let ret = [];
    stages.forEach((el, idx) => obj[el] = (obj[el] || 0) + 1);
    for(let i = 1; i <= N; i++) {
        if (obj[i]) {
            tmp = obj[i] / len;
            len -= obj[i];
            obj[i] = tmp;
        }
        else obj[i] = 0;
    }
    tmp = Object.entries(obj).sort((a, b) => b[1] - a[1]);
    tmp.forEach(el => {if (el[0] <= N) ret.push(Number(el[0]));});
    return (ret);
}
