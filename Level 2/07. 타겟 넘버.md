function solution(numbers, target) {
    const len = numbers.length;
    function calc(i, sum) {
        let ret = 0;
        if (i === len) ret += (sum === target);
        else {
            ret += calc(i + 1, sum + numbers[i]);
            ret += calc(i + 1, sum - numbers[i]);
        }
        return ret;
    }
    return calc(0, 0);
}

function solution(numbers, target) {
    let answer = 0;
    const dfs = (arr, aim) => {
        if (!arr.length && !aim) 
            return ++answer;
        else if (!arr.length && aim)
            return ;
        dfs(arr.slice(1), aim + arr[0]);
        dfs(arr.slice(1), aim - arr[0]);
    }
    dfs(numbers, target);
    return answer;
}
