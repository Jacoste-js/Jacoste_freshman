function solution(n) {
    let res = 0;
    for (let i = 1; n > 0; i *= 10, n = Math.floor(n / 3)) {
        if (n % 3) res += (n % 3) * i;
        else {
            res += 4 * i;
            n -= 1;
        }
    }
    return res.toString();
}

function solution(n) {
    let answer = "";
    let len = 0;
    let s = 0;
    for (let i = 1; ; i++) {
        if (n >= (Math.pow(3, i) - 1) / 2  && n < (Math.pow(3, i+1) - 1) / 2) {
            s = (Math.pow(3, i) - 1) / 2;
            len = i;
            break;
        }
    }
    for (let i = len; i > 0; --i) {
        let r = Math.pow(3, i - 1);
        if (n < s + r) {
            answer += "1";
        }
        else if (n < s + r * 2) {
            answer += "2";
            s += r;
        }
        else {
            answer += "4";
            s += r * 2;
        }
    }
    return answer;
}

function solution(n) {
    let arr = [];
    function rec(num) {
        if (num <= 3)
        {
            num % 3 === 0 ? arr.push('4') : arr.push(num)
            return ;
        }
        rec(Math.floor(num % 3 ? num / 3 : (num - 1) / 3));
        arr.push(num % 3 === 0 ? '4' : num % 3);
    }
    rec(n);
    return (arr.join(''));
}
