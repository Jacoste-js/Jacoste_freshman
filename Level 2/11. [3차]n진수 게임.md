function solution(n, t, m, p) {
    let res = '';
    let tmp = '';
    for (let i = 0; res.length < t; i++) {
        tmp += i.toString(n);
        if (tmp.length >= m) {
            res += tmp[p - 1]
            tmp = tmp.substr(m);
        }
    }
    return res.toUpperCase();
}

function solution(n, t, m, p) {
    let full = '';
    for (let i = 0; full.length < t * m; i++) {
        full += i.toString(n)
    }
    return full.split('')
                .filter((el, i) => i % m === p - 1)
                .slice(0, t)
                .join('')
                .toUpperCase();
}
