function solution(k, dungeons) {
    let max = dungeons.length;
    let answer = 0;
    const rec = (sum, arr) => {
        answer = Math.max(answer, max - arr.length);
        for (let i = 0; i < arr.length; i++) {
            if (sum - arr[i][0] >= 0)
                rec(sum - arr[i][1], [...arr.slice(0, i), ...arr.slice(i + 1)]);
        }
    }
    rec(k, dungeons);
    return answer;
}
