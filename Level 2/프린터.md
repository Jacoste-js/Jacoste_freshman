function solution(priorities, location) {
    let arr = priorities;
    let loc = location;
    let answer = 1;
    const getLoc = (idx , arrLen) => idx ? --idx : arrLen - 1;
    do {
        let max = Math.max(...arr);
        if (!loc && arr[0] === max)
            break;
        if (arr[0] !== max) {
            arr.push(arr.shift());
            loc = getLoc(loc, arr.length);
        } else {
            arr.shift();
            answer++;
            loc--;
        }
    } while (arr.length);
    return answer;
}
