function solution(arr) {
    const res = [0, 0];
    const compress = (x, y, l) => {
        const n = arr[x][y];
        let flag = 1;
        for (let i = x; i < x + l && flag; i++) {
            for (let j = y; j < y + l && flag; j++)
                if (arr[i][j] !== n) flag = 0;
        }
        if (flag) res[n] += 1;
        else {
            const h = l / 2;
            const a = [[x, y], [x + h, y], [x, y + h], [x + h, y + h]];
            for (let [nx, ny] of a) compress(nx, ny, h);
        }
    }
    compress(0, 0, arr.length);
    return res;
}
