function solution(arr) {
    const res = [0, 0];
    const compress = (x, y, l) => {
        const n = arr[x][y];
        let flag = 1;
        for (let i = x; i < x + l && flag; i++) {
            for (let j = y; j < y + l && flag; j++)
                if (arr[i][j] !== n) flag = 0;
        }
        if (flag) res[n] += 1;
        else {
            const h = l / 2;
            const a = [[x, y], [x + h, y], [x, y + h], [x + h, y + h]];
            for (let [nx, ny] of a) compress(nx, ny, h);
        }
    }
    compress(0, 0, arr.length);
    return res;
}

function solution(arr) {
    const answer = [0, 0];
    const quadTree = (piece) => {
        const sum = piece.flat().reduce((acc, cur) => acc + cur);
        if (!sum) answer[0]++;
        else if (sum === piece.length ** 2) answer[1]++;
        else if (piece.length > 1) {
            const quad = makeQuad(piece);
            for (let i = 0; i < 4; i++) quadTree(quad[i]);
        }
    }
    quadTree(arr);
    return answer;
}

function makeQuad (piece) {
    const quad = [[], [], [], []];
    const crtr = piece.length / 2;
    for (let i = 0; i < crtr * 2; i++) {
            if (i < crtr) {
                quad[0].push(piece[i].slice(0, crtr));
                quad[1].push(piece[i].slice(crtr));
            } else {
                quad[2].push(piece[i].slice(0, crtr));
                quad[3].push(piece[i].slice(crtr));
            }
    }
    return quad;
}

function solution(arr) {
    let ret = [0, 0];
    function rec(rawStart, rawEnd, colStart, colEnd) {
        if (rawStart === rawEnd || colStart === colEnd)
            return '' + arr[rawStart][colEnd];
        let q1 = rec(rawStart, (rawStart + rawEnd - 1) / 2, colStart, (colStart + colEnd - 1) / 2);
        let q2 = rec(rawStart, (rawStart + rawEnd - 1) / 2, (colStart + colEnd + 1) / 2, colEnd);
        let q3 = rec((rawStart + rawEnd + 1) / 2, rawEnd, colStart, (colStart + colEnd - 1) / 2);
        let q4 = rec((rawStart + rawEnd + 1) / 2, rawEnd, (colStart + colEnd + 1) / 2, colEnd);
        if (q1.length === 1 && q1 === q2 && q1 === q3 && q1 === q4)
            return '' + q1;
        return '' + q1 + q2 + q3 + q4;
    }
    arr = rec(0, arr.length - 1, 0, arr.length - 1)
    for (let i = 0; i < arr.length; i++)
        arr[i] === '0' ? ret[0]++ : ret[1]++
    return ret
}
