function solution(maps) {
    const gy = maps.length - 1
    const gx = maps[0].length - 1;
    const move = [[1, 0], [-1, 0], [0, 1], [0,-1]];
    const len = {'0, 0' : 1};
    const que = [[0, 0]];
    
    while (que.length) {
        const pos = que.shift();
        const [y, x] = pos;
        move.forEach((el) => {
            const my = y + el[0];
            const mx = x + el[1];
            if (!(mx > gx || mx < 0 || my > gy || my < 0)
               && maps[my][mx] === 1) {
                que.push([my, mx]);
                maps[my][mx] = 0;
                len[`${my}, ${mx}`] = len[`${y}, ${x}`] + 1;
            }
        })
    }
    return len[`${gy}, ${gx}`] === undefined ? -1 : len[`${gy}, ${gx}`];
}

function solution(maps) {
    const h = maps.length - 1;
    const w = maps[0].length - 1;
    const moveY = [+1, -1, 0, 0];
    const moveX = [0, 0, +1, -1];
    const visited = Array.from(new Array(h + 1), () => new Array(w + 1).fill(0));
    const path = Array.from(new Array(h + 1), () => new Array(w + 1).fill(0));
    const canVisit = (y, x) => (y >= 0 && y <= h && x >= 0 && x <= w) && maps[y][x] === 1 && !visited[y][x];
    const bfs = (y, x) => {
        const queue = [[0, 0]];
        path[y][x] = 1;
        
        while (queue.length) {
            const target = queue.shift();
            const currentY = target[0];
            const currentX = target[1];
            if (visited[currentY][currentX] === 1) continue;
            
            visited[currentY][currentX] = 1;
            for (let i = 0; i < 4; i++) {
                const nextY = currentY + moveY[i];
                const nextX = currentX + moveX[i];
                if (!canVisit(nextY, nextX)) continue;

                queue.push([nextY, nextX]);
                path[nextY][nextX] = path[currentY][currentX] + 1;
            }
        }
    }
    bfs(0, 0);
    return path[h][w] || -1;
}

function solution(maps) {
    let min = maps.length * maps[0].length;
    let obj = new Object();
    let check = 0;

    function rec(arr, cnt, x, y){
        if (check === 1 && min === maps.length + maps[0].length - 1)
            return ;
        if (obj['x' + x + 'y'+ y] <= cnt || min <= cnt)
            return ;
        else
            obj['x' + x + 'y'+ y] = cnt;
        if (x === maps[0].length - 1 && y === maps.length - 1 && cnt < min)
        {
            min = cnt + 1;
            check = 1;
            return ;
        }
        arr[y][x] = 0;
        if (y < arr.length - 1 && arr[y + 1][x] === 1)
            rec(arr, cnt + 1, x, y + 1);
        if (x < arr[0].length - 1 && arr[y][x + 1] === 1)
            rec(arr, cnt + 1, x + 1, y);
        if (y > 0 && arr[y - 1][x] === 1 && x < arr[0].length - 2)
            rec(arr, cnt + 1, x, y - 1);
        if (x > 0 && arr[y][x - 1] === 1 && y < arr.length - 2)
            rec(arr, cnt + 1, x - 1, y);
        arr[y][x] = 1;
    }
    rec(maps, 0, 0, 0);
    if (min === maps.length * maps[0].length && check === 0)
        min = -1;
    return min;
}
