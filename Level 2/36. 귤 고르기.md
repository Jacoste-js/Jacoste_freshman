function solution(k, tangerine) {
    const box = {};
    tangerine.forEach((el) => box[el] = (box[el] || 0) + 1);
    
    const arr = Object.values(box).sort((a, b) => b - a);
    for (let num = 0; num < arr.length; num++) {
        if (k <= 0) return num;
        k -= arr[num];
    }
    return arr.length;
}
