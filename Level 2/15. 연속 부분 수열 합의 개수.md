function solution(elements) {
    const sum = new Set();
    const el = elements.concat(elements);
    for (let i = 1; i <= elements.length; i++) {
        for (let j = 0; j < elements.length; j++) {
            sum.add(el.slice(j, j + i).reduce((a, b) => a + b, 0));
        }
    }
    return sum.size;
}
