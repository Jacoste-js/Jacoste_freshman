function solution(len, wei, trucks) {
    const bridge = Array.from({length: len}, () => 0);
    let time = 0;
    for (let on = 0; trucks.length; time++) {
        on -= bridge.shift();
        if (on + trucks[0] <= wei) {
            on += trucks[0];
            bridge.push(trucks.shift(trucks[0]));
        }
        else bridge.push(0);
    }
    return time + bridge.length;
}
