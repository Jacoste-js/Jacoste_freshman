function solution(dirs) {
    const point = new Set();
    let cur = [0, 0];
    let cnt = 0;
    
    const move = (des) => {
        if (des[0] > 5 || des[0] < -5 || des[1] > 5 || des[1] < -5)
            return ;
        if (!point.has(cur + ',' + des)) {
            point.add(cur + ',' + des);
            point.add(des + ',' + cur);
            cnt += 1;
        }
        cur = des;
    }
    
    for (let i = 0; i < dirs.length; i++) {
        switch (dirs[i]) {
            case 'U':
                move([cur[0], cur[1] + 1]);
                break;
            case 'D':
                move([cur[0], cur[1] - 1]);
                break;
            case 'R':
                move([cur[0] + 1, cur[1]]);
                break;
            case 'L':
                move([cur[0] - 1, cur[1]]);
                break;
        }
    }
    return cnt;
}

function solution(dirs) {
    const move = {U: [0, +1], D: [0, -1], R: [+1, 0], L: [-1, 0]};
    let last = [0, 0];
    let tmp = [];
    let x = 0;
    let y = 0;
    const checkExper = (needle, haystack) => {
        for (let i = 0; i < haystack.length; i++)
            if (JSON.stringify(needle) === JSON.stringify(haystack[i]) || 
                JSON.stringify(needle) === JSON.stringify([...haystack[i].slice(2), ...haystack[i].slice(0, 2)]))
                return false;
        return true;
    }
    const loc = dirs.split('').reduce((acc, cur, i) => {
        x = last[0] + move[cur][0];
        y = last[1] + move[cur][1];
        if (Math.abs(x) > 5 || Math.abs(y) > 5) return acc;
        tmp = [...last, x, y];
        last = [x, y];
        if (checkExper(tmp, acc)) acc.push(tmp);
        return acc;
    }, []);

    return loc.length;
}
