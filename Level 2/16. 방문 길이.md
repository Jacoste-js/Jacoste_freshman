function solution(dirs) {
    const point = new Set();
    let cur = [0, 0];
    let cnt = 0;
    
    const move = (des) => {
        if (des[0] > 5 || des[0] < -5 || des[1] > 5 || des[1] < -5)
            return ;
        if (!point.has(cur + ',' + des)) {
            point.add(cur + ',' + des);
            point.add(des + ',' + cur);
            cnt += 1;
        }
        cur = des;
    }
    
    for (let i = 0; i < dirs.length; i++) {
        switch (dirs[i]) {
            case 'U':
                move([cur[0], cur[1] + 1]);
                break;
            case 'D':
                move([cur[0], cur[1] - 1]);
                break;
            case 'R':
                move([cur[0] + 1, cur[1]]);
                break;
            case 'L':
                move([cur[0] - 1, cur[1]]);
                break;
        }
    }
    return cnt;
}
