function solution(word) {
    const key = ["A", "E", "I", "O", "U"];
    const dic = [];
    const rec = (str, cnt) => {
        if (cnt > 5) return;
        dic.push(str);
        key.forEach((el) => {
            rec(str + el, cnt + 1);
        })
    }
    rec("", 0);
    return dic.sort().indexOf(word);
}

function solution(word) {
    let str = ['A', 'E', 'I', 'O', 'U'];
    let set = [];
    let tmp;
    
    function rec (arr, w) {
        if (w.length >= arr.length)
            return ;
        for (let i = 0; i < arr.length; i++)
        {
            tmp = w;
            if (tmp.length < arr.length)
            {
                set.push(tmp += arr[i]);
                rec(arr, tmp);
            }
        }
    }
    rec(str, '');
    return set.indexOf(word) + 1;
}
