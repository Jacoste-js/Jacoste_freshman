function solution(word) {
    let str = ['A', 'E', 'I', 'O', 'U'];
    let set = [];
    let tmp;
    
    function rec (arr, w) {
        if (w.length >= arr.length)
            return ;
        for (let i = 0; i < arr.length; i++)
        {
            tmp = w;
            if (tmp.length < arr.length)
            {
                set.push(tmp += arr[i]);
                rec(arr, tmp);
            }
        }
    }
    rec(str, '');
    return set.indexOf(word) + 1;
}
