function solution(n) {
    const fibonacci = [0, 1];
    for (let i = 1; i <= n; i++) fibonacci.push((fibonacci.shift() + fibonacci[0]) % 1000000007);
    return fibonacci[1];
}

function solution(n) {
    let arr = [1, 1];
    for (let i = 2; i <= n; i++)
        arr.push((arr[i - 1] + arr[i - 2]) % 1000000007);
    return arr[arr.length - 1];
}
