function check(i, j, place) {
    let x = [-1, 0, 1];
    let y = [-1, 0, 1];
    for (a of x) {
        if(i + a < 0 || i + a > 4) continue;
        for (b of y) {
            if (j + b < 0 || j + b > 4) continue;
            if (a === 0 && b === 0) continue;
            if (place[i + a][j + b] === 'P' && a!==0 && b!==0) {
                if (place[i + a][j] !=='X' || place[i][j + b] !== 'X')
                    return false;
            }
            else if (place[i + a][j + b] === 'P')
                return false;
        }
    }
    let vec = [-2, 2];
    for (a of vec) {
        if (i + a < 0 || i + a > 4) continue;
        if (place[i + a][j] === 'P' && place[i + a / 2][j] !== 'X')
            return false;
        if (place[i][j + a] === 'P' && place[i][j + a / 2] !== 'X')
            return false;
    }
    return true;
}
function solution(places) {
    let answer = [1, 1, 1, 1, 1];
    places.forEach((place, placeIdx) => {
        let flag = true;
        for (let i = 0; i < 5; i++) {
            for (let j = 0; j < 5; j++){
               if (place[i][j] === 'P')
                   if (!check(i, j, place)) {
                       answer[placeIdx] = 0;
                       flag = false;
                       break;
                   }
            }
            if (flag === false) break;
        }
    })
    return answer;
}
