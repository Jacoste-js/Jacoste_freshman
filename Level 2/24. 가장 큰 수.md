function solution(numbers) {
    const compareStrNum = (a, b) => {
        return Number(b + a) - Number(a + b)
    }
    const answer = numbers.sort((a1, b1) => {
        return compareStrNum(a1.toString(), b1.toString());
    }).join("")
    return answer.replace(/(^0+)/, "") === "" ? "0" : answer;
}
