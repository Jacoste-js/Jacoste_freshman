function solution(n, k) {
    let arr = Array.from({length: n}, (_, i) => i + 1);
    let ret = [];
    let idx, fac;
    
    function factor (num) {
        let sum = 1;
        while (--num > 0) sum *= num;
        return sum;
    }
    
    for (let i = 0; i < n - 1; i++) {
        if (k === 0) k = fac;
        fac = factor(n - i);
        idx = Math.floor((k - 1) / fac);
        ret.push(arr[idx]);
        arr.splice([idx], 1);
        k %= fac;
    }
    
    ret.push(arr[0]);
    return ret;
}
